protected virtual bool ShouldAddInterceptionAttributes(ProxyMethodDefinition methodDefinition)
{
    return methodDefinition.InterceptorAttributes.Length > 0 && (methodDefinition.TypeDefinition.IsInterface || !methodDefinition.Method.IsVirtual);
}
